function binarySearch(arr, target):
  low  ← 0
  high ← arr.length - 1

  while low ≤ high:
    mid ← ⌊(low + high) / 2⌋

    if   arr[mid] == target → return mid
    elif arr[mid]  < target → low  ← mid + 1
    else                   → high ← mid - 1

  return -1  // not found
